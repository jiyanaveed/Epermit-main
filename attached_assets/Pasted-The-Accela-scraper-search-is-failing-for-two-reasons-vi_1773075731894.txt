// Clear End Date
const endDate = await page.$('input[id*="EndDate"]');
if (endDate && await endDate.isVisible().catch(() => false)) {
  await endDate.click({ clickCount: 3 });
  await endDate.press('Backspace');
  await page.waitForTimeout(300);
  console.log("  Cleared End Date");
}

// Clear Start Date  
const startDate = await page.$('input[id*="StartDate"]');
if (startDate && await startDate.isVisible().catch(() => false)) {
  await startDate.click({ clickCount: 3 });
  await startDate.press('Backspace');
  await page.waitForTimeout(300);
  console.log("  Cleared Start Date");
}

// Check "Search All Records"
const searchAll = await page.$('input[type="checkbox"]');
const allCheckboxes = await page.$$('input[type="checkbox"]');
for (const cb of allCheckboxes) {
  const id = await cb.getAttribute('id').catch(() => '');
  const nearby = await cb.evaluate(el => el.parentElement?.textContent?.trim() || '').catch(() => '');
  if (id.includes('SearchAll') || id.includes('chkSearchAll') || nearby.includes('Search All')) {
    const checked = await cb.isChecked().catch(() => false);
    if (!checked) {
      await cb.check();
      console.log("  Checked Search All Records");
      await page.waitForTimeout(1000);
    }
    break;
  }
}
